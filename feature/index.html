Per-Site Metrics

Embed snippet on each website to capture real page views 
<script>
(function() {
  const API = "https://abacus.jasoncameron.dev";
  const NS = "makaminski_site_views_v1";
  const SITE = "lace_luxx_com"; // or modular_equity_com / michael_kaminski_io
  const isAI = /bot|crawler|spider|gptbot|anthropic-ai|claudebot|perplexitybot|ccbot/i.test(navigator.userAgent);
  const viewerKey = "mk_site_viewer_id_" + SITE;
  const isNewHuman = !isAI && !localStorage.getItem(viewerKey);
  if (isNewHuman) localStorage.setItem(viewerKey, Date.now().toString(36));
  const hit = (k) => fetch(`${API}/hit/${NS}/${k}`).catch(() => null);
  hit(`site_${SITE}_views_total`);
  hit(isAI ? `site_${SITE}_views_ai` : `site_${SITE}_views_human`);
  if (isNewHuman) hit(`site_${SITE}_real_human_views`);
  hit("portfolio_views_total");
  hit(isAI ? "portfolio_views_ai" : "portfolio_views_human");
  if (isNewHuman) hit("portfolio_real_human_views");
})();
</script>

Tracking Notes

  • Real human page views are approximated using a browser-local viewer ID plus user-agent classification.
  • AI page views/clicks are classified by known crawler/agent signatures in user-agent strings.
  • Classification is heuristic-based and intended for directional analytics, not legal attribution.
